  function addPoints(teamName, pointsToAdd) {
    const team = teams.find(t => t.name === teamName);
    if (team) {
      team.points += pointsToAdd;
      updateLeaderboard();
    }
  }

  function updateLeaderboard() {
  teams.sort((a, b) => b.points - a.points);
  const table = document.getElementById('leaderboard');
  table.innerHTML = '<tr><th>Rank</th><th>Team Name</th><th>Branch</th><th>Points</th></tr>';

  let lastScore = -1;
  let lastRank = 0;
  let rankCounter = 0;

  teams.forEach((team) => {
    rankCounter++; // Increment for each team
    if (team.points !== lastScore) {
      lastRank = rankCounter;
      lastScore = team.points;
    }

    const row = table.insertRow(-1);
    const rankCell = row.insertCell(0);
    const nameCell = row.insertCell(1);
    const branchCell = row.insertCell(2);
    const pointsCell = row.insertCell(3);
    rankCell.innerHTML = lastRank;
    nameCell.innerHTML = team.name;
    branchCell.innerHTML = team.branch;
    pointsCell.innerHTML = team.points;

    if (lastRank === 1) {
        rankCell.classList.add('gold');
        nameCell.classList.add('gold');
        branchCell.classList.add('gold');
        pointsCell.classList.add('gold');
      } else if (lastRank === 2) {
        rankCell.classList.add('red');
        nameCell.classList.add('red');
        branchCell.classList.add('red');
        pointsCell.classList.add('red');
      } else if (lastRank === 3) {
        rankCell.classList.add('blue');
        nameCell.classList.add('blue');
        branchCell.classList.add('blue');
        pointsCell.classList.add('blue');
      }
    });
  }


  updateLeaderboard();
